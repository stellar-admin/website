Breadcrumbs provide navigation context and allow users to quickly navigate the current page's hierarchy.

<Demo src="/demo/tag-helpers/breadcrumb-intro.html">
  <include>./_include/breadcrumb-intro.mdx</include>
</Demo>

## Usage

```razor
<sui-breadcrumb>
    <sui-breadcrumb-list>
        <sui-breadcrumb-item>
            <sui-breadcrumb-link href="...">Level 1</sui-breadcrumb-link>
        </sui-breadcrumb-item>
        <sui-breadcrumb-separator/>
        <sui-breadcrumb-item>
            <sui-breadcrumb-link href="...">Level 2</sui-breadcrumb-link>
        </sui-breadcrumb-item>
        <sui-breadcrumb-separator/>
        <sui-breadcrumb-page>Current Page</sui-breadcrumb-page>
    </sui-breadcrumb-list>
</sui-breadcrumb>
```
