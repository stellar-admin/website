```razor
<div class="w-full rounded-md border">
    <sui-table>
        <sui-table-header>
            <sui-table-row>
                <sui-table-head class="w-[100px]">Booking #</sui-table-head>
                <sui-table-head>Status</sui-table-head>
                <sui-table-head>Destination</sui-table-head>
                <sui-table-head class="text-right">Amount</sui-table-head>
            </sui-table-row>
        </sui-table-header>
        <sui-table-body>
            @foreach (var booking in StaticData.Bookings)
            {
                <sui-table-row>
                    <sui-table-cell class="font-medium">@booking.Id</sui-table-cell>
                    <sui-table-cell>
                        <sui-badge variant="@GetBadgeVariant(booking.Status)">
                            @booking.Status.ToString()
                        </sui-badge>
                    </sui-table-cell>
                    <sui-table-cell>@booking.Destination</sui-table-cell>
                    <sui-table-cell class="text-right">
                        @booking.Amount.ToString("N")
                    </sui-table-cell>
                </sui-table-row>
            }
        </sui-table-body>
        <sui-table-footer>
            <sui-table-row>
                <sui-table-cell colspan="3">
                    Total Active Value
                </sui-table-cell>
                <sui-table-cell class="text-right">
                    @{
                        var activeValue = StaticData.Bookings
                            .Where(b => b.Status != BookingStatus.Cancelled)
                            .Select(b => b.Amount)
                            .Sum();
                    }
                    @activeValue.ToString("N")
                </sui-table-cell>
            </sui-table-row>
        </sui-table-footer>
    </sui-table>
</div>

@functions
{
    BadgeVariant GetBadgeVariant(BookingStatus status) => status switch
    {
        BookingStatus.Cancelled => BadgeVariant.Destructive,
        BookingStatus.Pending => BadgeVariant.Secondary,
        _ => BadgeVariant.Default
    };
}
```
